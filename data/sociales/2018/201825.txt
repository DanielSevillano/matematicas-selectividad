    -1 & 2 \\
    -3 & 4
\end{pmatrix}, \quad
B = \begin{pmatrix}
    -1 & 2 & 1 \\
    3 & 0 & 2
\end{pmatrix} \quad \text{y} \quad
C = \begin{pmatrix}
    3 & 0 & 1 \\
    2 & -1 & -1
\end{pmatrix}.$$
